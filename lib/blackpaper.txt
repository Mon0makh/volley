Iterable<PopupMenuItem> _popupButtonsElement(Map<String, String> val) sync* {
  for (var item in val.entries) {
    yield PopupMenuItem(
      value: item.key,
      child: TextButton(
        onPressed: null,
        child: Text(
          item.key,
          style: const TextStyle(
            color: Colors.black,
            fontSize: 13,
            fontWeight: FontWeight.w700,
          ),
        ),
      ),
    );
  }
}
